function sortArrayWithoutChangingNumbers(array, numbersToKeep) {
  const numberIndices = numbersToKeep.map((number) => array.indexOf(number));

  // Remove the numbers from the array
  numbersToKeep.forEach((number) => {
    const index = array.indexOf(number);
    if (index !== -1) {
      array.splice(index, 1);
    }
  });

  // Sort the remaining elements of the array
  array.sort((a, b) => a - b);

  // Insert the numbers back into the array at their original positions
  numberIndices.forEach((index, i) => {
    array.splice(index, 0, numbersToKeep[i]);
  });

  return array;
}

// Example usage
const numbers = [5, 2, 9, 1, 7, 3];
const numbersToKeep = [9, 3, 5];

const sortedArray = sortArrayWithoutChangingNumbers(numbers, numbersToKeep);
console.log(sortedArray); // Output: [1, 2, 5, 9, 7, 3]
